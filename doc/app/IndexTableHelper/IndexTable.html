def add_column *args
        column = {
                :name => :name,
                :header => "Some name"
        }.merge(args.extract_options!)

        # Remember to pass something to return as value in table cell
        unless column[:method]
                raise ArgumentError.new "You have to pass :method => :some_thing"
        end

        @columns << column
end
